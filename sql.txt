CREATE DATABASE crud_escolar
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'Portuguese_Brazil.1252'
    LC_CTYPE = 'Portuguese_Brazil.1252'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1;

//TABELAS
CREATE TABLE public.pessoa
(
    pes_id numeric(6,0) NOT NULL,
    pes_matricula character varying(30) COLLATE pg_catalog."default" NOT NULL,
    pes_cpf character varying(11) COLLATE pg_catalog."default" NOT NULL,
    pes_rg character varying(9) COLLATE pg_catalog."default" NOT NULL,
    pes_pnome character varying(15) COLLATE pg_catalog."default" NOT NULL,
    pes_unome character varying(50) COLLATE pg_catalog."default" NOT NULL,
    pes_email character varying(255) COLLATE pg_catalog."default" NOT NULL,
    pes_data_nascimento date NOT NULL,
    pes_dtcadastro timestamp with time zone NOT NULL,
    pes_end_id numeric(6,0) NOT NULL,
    CONSTRAINT pessoa_pkey PRIMARY KEY (pes_id),
    CONSTRAINT pes_cpf UNIQUE (pes_cpf),
    CONSTRAINT pes_matricula UNIQUE (pes_matricula),
    CONSTRAINT pes_rg UNIQUE (pes_rg)
)

TABLESPACE pg_default;

ALTER TABLE public.pessoa
    OWNER to postgres;

CREATE TABLE public.endereco
(
    end_id numeric(6,0) NOT NULL,
    end_cep character varying(8) COLLATE pg_catalog."default" NOT NULL,
    end_estado character varying(50) COLLATE pg_catalog."default" NOT NULL,
    end_cidade character varying(50) COLLATE pg_catalog."default" NOT NULL,
    end_numero numeric(8,0) NOT NULL,
    end_logradouro character varying(255) COLLATE pg_catalog."default" NOT NULL,
    end_complemento character varying(255) COLLATE pg_catalog."default",
    end_dtcadastro timestamp with time zone NOT NULL,
    CONSTRAINT endereco_pkey PRIMARY KEY (end_id)
)

TABLESPACE pg_default;

ALTER TABLE public.endereco
    OWNER to postgres;

CREATE TABLE public.cursos
(
    cur_id numeric(6,0) NOT NULL,
    cur_nome character varying(30) COLLATE pg_catalog."default" NOT NULL,
    cur_nivel character varying(255) COLLATE pg_catalog."default" NOT NULL,
    cur_duracao numeric(2,0) NOT NULL,
    cur_mensalidade numeric(8,2) NOT NULL,
    cur_dtcadastro timestamp with time zone NOT NULL,
    CONSTRAINT cursos_pkey PRIMARY KEY (cur_id),
    CONSTRAINT cur_nome UNIQUE (cur_nome)
)

TABLESPACE pg_default;

ALTER TABLE public.cursos
    OWNER to postgres;
